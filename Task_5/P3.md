#include <stdio.h>
#include <string.h>

int main() {
    char str[100], temp;
    int i, len;

    // Take string input (until space or newline)
    printf("Enter the string  : ");
    scanf("%s", str);  // reads input until first whitespace

    // Get the length of the string
    len = strlen(str);

    // Reverse the string in-place
    for (i = 0; i < len / 2; i++) {
        temp = str[i];
        str[i] = str[len - i - 1];
        str[len - i - 1] = temp;
    }

    // Print the reversed string
    printf("Reverse string is : %s\n", str);

    return 0;
}
